const BASE = ''; const OC = 'dohun2026'; // 법제처 API 키 직접 입력 async function doSearch() { const kw = document.getElementById('searchInput').value.trim(); if (!kw) return; currentKeyword = kw; openCardIdx = null; const btn = document.getElementById('searchBtn'); btn.disabled = true; btn.textContent = '검색 중…'; showLoading(); try { // 법제처 직접 호출 (JSONP 방식) const data = await fetchLawDirect(kw); const laws = [].concat(data?.LawSearch?.law || []); lawData = laws.map(l => ({ id: l.법령ID||'', name: l.법령명한글||l.법령명||'(법령명 없음)', type: classifyType(l.법령구분명||''), date: l.공포일자||'', msn: l.법령일련번호||'' })); renderResults(lawData, kw); } catch(e) { showError(`검색 실패: ${e.message}`); } finally { btn.disabled = false; btn.textContent = '검색'; } } function fetchLawDirect(query) { return new Promise((resolve, reject) => { const cbName = 'lawCb_' + Date.now(); const script = document.createElement('script'); script.src = `https://www.law.go.kr/DRF/lawSearch.do?OC=${OC}&target=law&type=JSON&display=20&query=${encodeURIComponent(query)}&callback=${cbName}`; window[cbName] = (data) => { delete window[cbName]; document.body.removeChild(script); resolve(data); }; script.onerror = () => reject(new Error('법제처 연결 실패')); document.body.appendChild(script); }); }
